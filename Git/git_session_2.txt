#Branching and merging
    git branch
    git branch <BranchName>
    git checkout <BranchName>
    git branch -d <BranchName>      -D fore delete
    git checkout -b <BranchName>
    git branch -m <NewBranchName>
    git checkout master
    git merge <BranchName>
    git push origin <BranchName>

#Mastring stash
    git stash
    git stash list

    git stash list
    git stash apply

#Restore and clear
    untrack file file
    git restore file
    git restore --staged file
    git add .
    git restore --staged
    git clean -n    show files need to beleted
    git clean -f    delete files

#Reseting the Head
    git log
    git log --oneline
    git show <commit id>
    git diff <commit id>..HEAD
    git commit --amend      modify commit message
    git reset <commit id>
    git reset --hard HEAD~1
    git reflog
    git reset --hard HEAD@{1}
    git reset --hard <commit id>
    git push origin main --force

#gitignore

#Tags and Releases
    git tag

    git push origin v1.0.0

    git tag -a v2.0.0. -m "secend version"
    git push origin v2.0.0

    git tag -l "v1.*"
    git tag -d "v1.0.0"

    git push origin --delete v1.0.0
